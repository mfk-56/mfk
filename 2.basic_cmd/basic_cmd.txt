ls	查看（指定）目录下的文件

cd	向指定目录切换

chmod	修改文件权限

file	查看文件信息

pwd	显示当前工作路径

mkdir	新建目录（文件夹）

cp	拷贝文件

mv	剪切（移动）文件 / 重命名

rm	删除文件

exit	退出用户

touch	新建文件

cat	以文本方式显示文件内容

ifconfig查看网络信息

echo	输出

ps	查看当前正在运行的进程信息(-ef)

tar	压缩、解压缩
	压缩使用参数        c
	解压缩使用参数      x
	显示压缩/解压缩过程 v
	参数z - gzip压缩包格式
	参数j - bzip压缩包格式

	压缩指令格式  ：tar 压缩参数f 生成的压缩包名 要压缩的文件(多个 用空格隔开)
	解压缩指令格式：tar 解压参数f 要解压的压缩包名 ( -C 指定解压路径 )
(解压缩zip格式压缩文件 使用指令 unzip即可)

find	查找文件
	find -name "xxxx"

top	linux任务管理器界面
